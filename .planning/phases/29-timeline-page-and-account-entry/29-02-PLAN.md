const displayPoints = computed<MapPointDisplay[]>(() => {
  const latestByPlaceId = new Map<string, TravelRecord>()
  for (const record of travelRecords.value) {
    const existing = latestByPlaceId.get(record.placeId)
    if (!existing || record.createdAt > existing.createdAt) {
      latestByPlaceId.set(record.placeId, record)
    }
  }
})

  <action>
    新建 `apps/web/src/services/timeline.ts`，实现时间轴纯函数与类型：

    1. 导出 `TimelineEntry` 类型，至少包含：
       - `recordId`
       - `placeId`
       - `displayName`
       - `parentLabel`
       - `subtitle`
       - `typeLabel`
       - `startDate`
       - `endDate`
       - `createdAt`
       - `hasKnownDate`
       - `sortDate`
       - `visitOrdinal`
       - `visitCount`

    2. 导出 `buildTimelineEntries(records: TravelRecord[]): TimelineEntry[]`，规则固定如下：
       - 每条 record 生成一条 entry，绝不按 `placeId` 合并
       - `hasKnownDate = startDate !== null`
       - 已知日期 entry 的 `sortDate = endDate ?? startDate`
       - 未知日期 entry 的 `sortDate = null`
       - 同一 `placeId` 的 entry 先按全局排序规则排序，再生成 `visitOrdinal = 1..N`，`visitCount = N`

    3. 全局排序必须精确实现：
       - 先按 `hasKnownDate` 排序：`true` 在前，`false` 在后
       - 已知日期之间按 `sortDate` 升序（最早优先）
       - 若 `sortDate` 相同，再按 `startDate ?? ''` 升序
       - 若仍相同，再按 `createdAt` 升序
       - 若仍相同，再按 `id` 升序
       - 未知日期记录只允许用 `createdAt` / `id` 做稳定性排序，不能把它渲染为“旅行日期”

    4. 新建 `apps/web/src/services/timeline.spec.ts`，至少覆盖这些固定场景：
       - `keeps multiple visits for the same place as separate timeline entries`
       - `orders dated entries from earliest to latest`
       - `places unknown-date entries after dated entries`
       - `assigns visitOrdinal and visitCount per place`
  </action>
