    async getTravelStats(userId: string): Promise<TravelStatsResponse> {
      const [totalTrips, uniquePlaceRecords] = await Promise.all([
        this.prisma.userTravelRecord.count({ where: { userId } }),
        this.prisma.userTravelRecord.findMany({
          where: { userId },
          select: { placeId: true },
          distinct: ['placeId'],
        }),
      ])
      return { totalTrips, uniquePlaces: uniquePlaceRecords.length }
    }

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| HTTP client → GET /records/stats | 未认证请求必须被拒绝，不能泄露任何用户数据 |
| session cookie → userId | userId 必须从 session 取，不能从请求参数读取 |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-30-01 | Elevation of Privilege | GET /records/stats | mitigate | `@UseGuards(SessionAuthGuard)` 装饰器，未认证返回 401 |
| T-30-02 | Information Disclosure | RecordsController.getStats | mitigate | `@CurrentUser()` 从 session 取 userId，不接受客户端传入 userId 参数 |
| T-30-03 | Information Disclosure | RecordsRepository.getTravelStats | accept | 查询仅返回聚合数值（count），不返回任何 PII 字段 |
</threat_model>
